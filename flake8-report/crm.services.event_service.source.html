37 def assign_support(self, event, user):
38 """
39 Assigne ou retire un support manager d'un événement.
40 """

44

47

49 event.users.remove(user)

51 event.users.append(user)

55  
56 except ValueError as e:
57 log_error(f"Erreur de validation : {str(e)}")
58 raise
59 except Exception as e:
60 log_error(

63 self.repository.session.rollback()

90 def update(self, event_id, new_data):
91 """
92 Met à jour un événement et gère l'affectation des supports.
93 """
94 try:
95 event = self.get_by_id(event_id)
96 if not event:
97 raise ValueError(f"L'événement {event_id} n'existe pas.")
98  
99 support_id = new_data.pop("support_id", None) # Récupère l'ID du support
100  
101 # Mise à jour des autres champs
102 for key, value in new_data.items():
103 setattr(event, key, value)
104  
105 # Gère l'affectation du support avec assign_support()
106 if support_id is not None:
107 support_user = self.user_repo.get_by_id(support_id)
108 self.assign_support(event, support_user)
109  
110 self.repository.session.commit() # Sauvegarde des modifications
111  
112 return event
113 except Exception as e:
114 log_error(
